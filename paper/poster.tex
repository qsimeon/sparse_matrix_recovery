% ═══════════════ COLUMN 1 — Problem + Method ═══════════════
\begin{column}{0.32\linewidth}

\begin{tcolorbox}[posterbox,title=Problem]
\large
\textbf{Goal}: Recover full $N{\times}N$ connectivity matrix $W$ from \textbf{partial measurements} across multiple recording sessions.

\vspace{6mm}
\textbf{Challenge}: In small brain circuits (\textit{C.\ elegans}, zebrafish, organoids), we can only observe a \emph{subset} of neurons per session.

\vspace{6mm}
\textbf{Dynamical model}:
\[
  x_{t+1} = W\phi(x_t) + b_t
\]

Each session $k$ observes $\mathcal{M}_k \subset \{1,\ldots,N\}$ and stimulates $\mathcal{S}_k$.

\vspace{6mm}
\textbf{Key insight}: Pairwise covariance statistics can be \textbf{accumulated across sessions} whenever neurons $i,j$ are co-observed.
\end{tcolorbox}

\vspace{1cm}

\begin{tcolorbox}[posterbox,title=Method]
\large
\textbf{Step 1 --- Covariance accumulation} over $K{=}50$ sessions:
\[
  \hat{\Sigma}_{x_{t+1},x_t}(i,j) = \frac{1}{n_{ij}} \sum_{k: i,j\in\mathcal{M}_k} \hat{\Sigma}^{(k)}(i,j)
\]

\vspace{4mm}
\textbf{Step 2 --- Linear estimator}:
\[
  \hat{W} = \hat{\Sigma}_{x_{t+1},x_t}\, \hat{\Sigma}_{x_t,x_t}^{-1}
\]
with diagonal zeroed ($W_{ii}{=}0$, no autapses).

\vspace{4mm}
\textbf{Step 3 --- Granger-causality refinement}: Projected gradient descent enforcing sparsity, non-negativity ($W_{ij} \geq 0$), and no autapses ($W_{ii} = 0$).

\vspace{4mm}
\textbf{Identifiability}: $K \geq \log(N^2/\delta)/p^2$ sessions suffice for all-pairs coverage with probability $\geq 1{-}\delta$.
\end{tcolorbox}

% ═══════════════ COLUMN 3 — Discoveries + Impact ═══════════════
\begin{column}{0.32\linewidth}

\begin{tcolorbox}[posterbox,title=Implicit Regularization (E6),colframe=neuripsred,colbacktitle=neuripsred]
\centering
\includegraphics[width=0.95\linewidth]{figures/fig10_oracle_comparison.pdf}
\\[4mm]
\large The ``incorrect'' linear approximation \textbf{outperforms the oracle} with known nonlinearity at \emph{all} tested regimes --- a James--Stein phenomenon.

\vspace{4mm}
\textbf{Why?}\; $\Sigma_{\phi(x),x} = D\Sigma_{x,x}$ where $\tanh$ compression makes $D$ ill-conditioned, amplifying noise in the oracle inversion.

\vspace{4mm}
{\color{neuripsred}\textbf{Practical implication:}} No need to characterize neuronal nonlinearity --- the simpler model is \emph{better}.
\end{tcolorbox}

\vspace{1cm}

\begin{tcolorbox}[posterbox,title=Error Decomposition]

\vspace{1cm}

\begin{tcolorbox}[posterbox,title=Practical Guidelines for Experimentalists,colframe=neuripsgreen,colbacktitle=neuripsgreen]
\large
\begin{enumerate}
  \item[\color{neuripsgreen}\textbf{1.}] \textbf{Measurement}: $\geq 50\%$ neuron coverage per session
  \item[\color{neuripsgreen}\textbf{2.}] \textbf{Stimulation}: Moderate ($\sigma \approx 0.5$--$1.0$) on $\sim 33\%$ of neurons
  \item[\color{neuripsgreen}\textbf{3.}] \textbf{Sessions}: $K \geq 50$ with random masks for all-pairs coverage
  \item[\color{neuripsgreen}\textbf{4.}] \textbf{Model}: Use the linear approximation --- simpler \emph{and} better
  \item[\color{neuripsgreen}\textbf{5.}] \textbf{Next}: Model CPG dynamics to tackle the $3.1\times$ dominant error
\end{enumerate}
\end{tcolorbox}

\begin{column}{0.32\linewidth}

% ── Problem ──
\begin{tcolorbox}[posterbox,title=Problem]
\large
You want to map how neurons are wired --- the connectivity matrix $W$ --- but you can \textbf{never record every neuron at once}.

\vspace{6mm}
In small brain circuits (\textit{C.\ elegans}, zebrafish, organoids), each recording session observes a different \emph{subset} of neurons. Different days, different views of the same circuit.

\vspace{6mm}
\textbf{Dynamical model}:
\[
  x_{t+1} = W\phi(x_t) + \underbrace{b_t^{\text{stim}}}_{\text{you control}} + \underbrace{b_t^{\text{CPG}}}_{\text{intrinsic}}
\]

\vspace{4mm}
\textbf{Key insight}: Whenever neurons $i$ and $j$ are co-observed, their covariance contributes to estimating $W_{ij}$. Over many sessions, you piece together the full picture.
\end{tcolorbox}

\vspace{1cm}

% ── Method ──
\begin{tcolorbox}[posterbox,title=Method]
\large
\textbf{Step 1 --- Covariance accumulation} over $K{=}50$ sessions:
\[
  \hat{\Sigma}_{x_{t+1},x_t}(i,j) = \frac{1}{n_{ij}} \sum_{k: i,j\in\mathcal{M}_k} \hat{\Sigma}^{(k)}(i,j)
\]

\vspace{4mm}
\textbf{Step 2 --- Linear estimator}:
\[
  \hat{W} = \hat{\Sigma}_{x_{t+1},x_t}\, \hat{\Sigma}_{x_t,x_t}^{-1}
\]
with diagonal zeroed ($W_{ii}{=}0$, no autapses).

\vspace{4mm}
\textbf{Step 3 --- Granger-causality refinement}: Projected gradient descent enforcing:
\begin{itemize}
  \item[\color{neuripsgreen}\textbullet] Sparsity (zero non-causal edges)
  \item[\color{neuripsgreen}\textbullet] Non-negativity ($W_{ij} \geq 0$)
  \item[\color{neuripsgreen}\textbullet] No autapses ($W_{ii} = 0$)
\end{itemize}

\vspace{4mm}
\textbf{Identifiability}: $K \geq \log(N^2/\delta)/p^2$ sessions suffice for all-pairs coverage with probability $\geq 1{-}\delta$.
\end{tcolorbox}

% ── Oracle Comparison ──
\begin{tcolorbox}[posterbox,title=Implicit Regularization (E6),colframe=neuripsred,colbacktitle=neuripsred]
\centering
\includegraphics[width=0.95\linewidth]{figures/fig10_oracle_comparison.pdf}
\\[4mm]
\large The ``wrong'' linear approximation \textbf{outperforms the oracle} that knows the true nonlinearity --- at \emph{every} stimulation level tested (up to $4.4\times$ worse for the oracle).

\vspace{4mm}
\textbf{Why?}\; Stein--Price identity: $\Sigma_{\phi(x),x} = D\Sigma_{x,x}$ where $D{=}\mathrm{diag}(\mathbb{E}[\mathrm{sech}^2(x_i)])$.
$\tanh$ compresses high-variance neurons more, making $D$ heterogeneous and the oracle's inversion ill-conditioned. The linear estimator is biased but better-conditioned --- a classic \textbf{bias-variance tradeoff}.

\vspace{4mm}
{\color{neuripsred}\textbf{For experimentalists:}} Don't bother characterizing the neuronal transfer function --- the simpler model is \emph{provably better}.
\end{tcolorbox}

\vspace{1cm}

\large

\vspace{4mm}

$\Rightarrow$ Future gains from modeling \textbf{intrinsic dynamics}, not refining the nonlinear approximation.
\end{tcolorbox}

\vspace{1cm}

% ── Practical Guidelines ──
\begin{tcolorbox}[posterbox,title=Practical Guidelines for Experimentalists,colframe=neuripsgreen,colbacktitle=neuripsgreen]
\large
\begin{itemize}\setlength{\itemsep}{2pt}
  \item[\color{neuripsgreen}\textbf{1.}] \textbf{Measurement}: $\geq 50\%$ coverage per session (plateaus above this)
  \item[\color{neuripsgreen}\textbf{2.}] \textbf{Stimulation}: Moderate ($\sigma \approx 0.5$--$1.0$) on $\sim 33\%$ of neurons
  \item[\color{neuripsgreen}\textbf{3.}] \textbf{Sessions}: $K \geq 50$ with random masks for all-pairs coverage
  \item[\color{neuripsgreen}\textbf{4.}] \textbf{Model}: Use the linear approximation --- simpler \emph{and} better
  \item[\color{neuripsgreen}\textbf{5.}] \textbf{Next}: Model CPG dynamics to tackle the $3.3\times$ dominant error
\end{itemize}
\end{tcolorbox}
